int pbuttonPin = 2;
int relayPin = 10;
int val = 0;
int prevState = 0; // To store the previous state of the button
int pushed = 0;
void setup() {
 pinMode(pbuttonPin, INPUT_PULLUP); // Internal pull-up resistor
 pinMode(relayPin, OUTPUT); // Set relay pin as output
 digitalWrite(relayPin, HIGH); // Turn relay off initially
}
void loop() {
 val = digitalRead(pbuttonPin); // Read the button state
 // Detect a button press event (transition from LOW to HIGH)
 if (val == LOW && prevState == HIGH) {
 pushed = 1 - pushed; // Toggle the pushed variable
 delay(100); // Debouncing delay
 }

 prevState = val; // Update the previous state
 // Control the relay based on the pushed state
 if (pushed == 1) {
 digitalWrite(relayPin, LOW); // Turn the relay on
 } else {
 digitalWrite(relayPin, HIGH); // Turn the relay off
 }
}
