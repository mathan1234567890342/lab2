stack = []
max_size = 5

while True:
    print("\n1. Push")
    print("2. Pop")
    print("3. Exit")

    choice = int(input("Enter your choice: "))

    if choice == 1:
        if len(stack) == max_size:
            print("Stack is full")
        else:
            value = input("Enter value: ")
            stack.append(value)
            print("Stack after push:", stack)

    elif choice == 2:
        if len(stack) == 0:
            print("Stack is empty")
        else:
            removed = stack.pop()
            print("Popped element:", removed)
            print("Stack after pop:", stack)

    elif choice == 3:
        break

    else:
